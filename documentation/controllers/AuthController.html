import {Body, Controller, Get, HttpException, HttpStatus, Post, Request, Query, Res, UseGuards} from '@nestjs/common';
import * as jwt from 'jsonwebtoken';
import {AuthUserDtoDto} from './dto/AuthUserDto.dto';
import * as CryptoJS from 'crypto-js';
import {Response} from "express";
import { WinstonLogger } from '@payk/nestjs-winston';
import { AuthService } from './auth.service';
import * as config from "config";
import { AuthGuard } from '@nestjs/passport';

@Controller('login')
export class AuthController {
    private readonly logger = new WinstonLogger(AuthController.name);
    constructor(private authService: AuthService) {}

    @UseGuards(AuthGuard('local'))
    @Post()
    async authWeb(@Res() res: Response,@Request() req){
        return res.status(HttpStatus.OK).json({token:req.user})
    }

    @Get()
    async authRedirect(@Query() query,@Res() res: Response){
        let decrypted = CryptoJS.AES.decrypt(query.user.replace(" ","+"),config.get("keys.aesKey"));
        let decryptVal = decrypted.toString(CryptoJS.enc.Utf8).split("#");
        let timeOutError = setTimeout(function (){
            throw new HttpException('Tiempo de respuesta excedido.', HttpStatus.REQUEST_TIMEOUT);
        },config.get("globals.timeOut"))
            let rolesSoap:any = await this.authService.getRoleSoap(query.app_uniqueid,decryptVal[0]);
            clearTimeout(timeOutError)
            if(rolesSoap.result){
                let token = jwt.sign({
                    permissions: rolesSoap.data,
                }, config.get("keys.jwtKey"), {
                    expiresIn: config.get("globals.expJWT")
                });
                return res.status(rolesSoap.code).json({token:token})
            }else{
                throw new HttpException('Error de Servidor', rolesSoap.code);
            }


    }


}
